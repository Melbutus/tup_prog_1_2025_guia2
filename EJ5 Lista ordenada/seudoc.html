Proceso lista_ordenada
    definir nom1,nom2,nom3 como texto;
    definir not1,not2,not3 como real;
    
    escribir'Ingresar el primer nombre y la nota:';
    leer nom1,not1;
    escribir'Ingresar el segundo nombre y la nota:';
    leer nom2,not2;
    escribir'Ingresar el tecer nombre y la nota:';
    leer nom3,not3 ;
    
    si not1>not2 y not2>not3 Entonces
        escribir nom1,not1;
        escribir nom2,not2;
        escribir nom3,not3;
    FinSi
    si not1>not3 y not3>not2 Entonces
        escribir nom1,not1;
        escribir nom3,not3;
        escribir nom2,not2;
    FinSi
    si not2>not1 y not1>not3 Entonces
        escribir nom2,not2;
        escribir nom1,not1;
        escribir nom3,not3;
    FinSi
    si not2>not3 y not3>not1 Entonces
        escribir nom2,not2;
        escribir nom3,not3;
        escribir nom1,not1;
    FinSi
    si not3>not1 y not1>not2 Entonces
        escribir nom3,not3;
        escribir nom1,not1;
        escribir nom2,not2;
    FinSi
    si not3>not2 y not2>not1 Entonces
        escribir nom3,not3;
        escribir nom2,not2;
        escribir nom1,not1;
    FinSi
FinProceso
